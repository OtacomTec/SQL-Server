--INICIO CABECALHO

declare @Nomenclatura as nvarchar(20)
declare @Observacao as nvarchar(100)
declare @Data as SMALLDATETIME

SET @Nomenclatura = 'SCP2006-04-28a'
SET @Observacao = 'Setando campos Scoreun e Scorecx para 0 na TBProdutos'	
SET @Data = CURRENT_TIMESTAMP


---LIMPEZA DA TABELA

DELETE FROM TBSCRIPT

---INSERCAO DO REGISTRO
INSERT INTO TBSCRIPT(DFNome_TBScript,DFObjetivo_TBScript,DFData_execucao_TBScript)
VALUES(@Nomenclatura,@Observacao,@Data)

---FIM CABECALHO

UPDATE TBProduto
SET DFScoreun_TBProduto = 0    
WHERE DFScoreun_TBProduto IS NULL

UPDATE TBProduto
SET DFScorecx_TBProduto = 0    
WHERE DFScorecx_TBProduto IS NULL

COMMIT TRANSACTION
