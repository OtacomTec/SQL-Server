--INICIO CABECALHO

declare @Nomenclatura as nvarchar(20)
declare @Observacao as nvarchar(100)
declare @Data as SMALLDATETIME

SET @Nomenclatura = 'SCP2007-09-11c'
SET @Observacao = 'Cria Tabela Separador' 
SET @Data = CURRENT_TIMESTAMP

---LIMPEZA DA TABELA

DELETE FROM TBSCRIPT

---INSERCAO DO REGISTRO

INSERT INTO BDRetaguarda.dbo.TBSCRIPT(DFNome_TBScript,DFObjetivo_TBScript,DFData_execucao_TBScript)
VALUES(@Nomenclatura,@Observacao,@Data)

---FIM CABECALHO

Create table [TBSeparador]
(
	[PKCodigo_TBSeparador] Integer NOT NULL,
	[DFDescricao_TBSeparador] Nvarchar(40) NULL,
	[DFSeparador_conferente_ambos_TBSeparador] bit NULL,
	[DFIntegrado_portal_TBSeparador] Bit NULL,
	[DFIntegrado_filiais_TBSeparador] Bit NULL,
	[DFData_alteracao_TBSeparador] Smalldatetime NULL,
	[DFCodigo_Identificador_TBSeparador] Integer NULL,
Primary Key ([PKCodigo_TBSeparador])
) 

GO
